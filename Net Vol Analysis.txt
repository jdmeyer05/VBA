Option Explicit

Sub analysis()

Dim a, b, c, d, e, f, g, z, rng As Long
Dim ws As Worksheet

rng = ThisWorkbook.Worksheets("sheet2").Cells(Rows.Count, "a").End(xlUp).Row

Set ws = ThisWorkbook.Worksheets("Sheet2")

a = 3
c = 5
b = 4
d = 6
e = 7
f = 8
    
For z = 2 To rng    'SETS LOOP FOR ROW MOVEMENT
    
    If ws.Cells(z, a).Value <> 0 _
    And (ws.Cells(z, b).Value = 0 _
    Or ws.Cells(z, c).Value = 0 _
    Or ws.Cells(z, d).Value = 0 _
    Or ws.Cells(z, e).Value = 0 _
    Or ws.Cells(z, f).Value = 0) Then
        ws.Cells(z, b).EntireRow.Interior.ColorIndex = 44  'HIGHLIGHTS LATEST PERIOD WITH ERROR
    End If
    
    If ws.Cells(z, b).Value <> 0 _
    And (ws.Cells(z, c).Value = 0 _
    Or ws.Cells(z, d).Value = 0 _
    Or ws.Cells(z, e).Value = 0 _
    Or ws.Cells(z, f).Value = 0) Then
        ws.Cells(z, b).EntireRow.Interior.ColorIndex = 44  'HIGHLIGHTS LATEST PERIOD WITH ERROR
    End If
    
    If ws.Cells(z, c).Value <> 0 _
    And (ws.Cells(z, d).Value = 0 _
    Or ws.Cells(z, e).Value = 0 _
    Or ws.Cells(z, f).Value = 0) Then
        ws.Cells(z, b).EntireRow.Interior.ColorIndex = 44  'HIGHLIGHTS LATEST PERIOD WITH ERROR
    End If
    
    If ws.Cells(z, d).Value <> 0 _
    And (ws.Cells(z, e).Value = 0 _
    Or ws.Cells(z, f).Value = 0) Then
        ws.Cells(z, b).EntireRow.Interior.ColorIndex = 44  'HIGHLIGHTS LATEST PERIOD WITH ERROR
    End If
    
    If ws.Cells(z, e).Value <> 0 _
    And (ws.Cells(z, f).Value = 0) Then
        ws.Cells(z, b).EntireRow.Interior.ColorIndex = 44  'HIGHLIGHTS LATEST PERIOD WITH ERROR
    End If

    
Next z

End Sub
